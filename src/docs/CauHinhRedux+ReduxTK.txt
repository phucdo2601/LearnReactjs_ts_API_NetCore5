- file store.ts:
-------------------------------------------------------------------------
        import { configureStore } from "@reduxjs/toolkit";
        import ContactInfoSlice from "../slices/contactInfoSlice";

        const reducer = {
        contacts: ContactInfoSlice,
        };

        const store = configureStore({
        reducer: {
            contacts: ContactInfoSlice,
        },
        devTools: true,
        });

        type StoreType = typeof store;
        export type AppDispatch = StoreType['dispatch'];

        type Config = {
        dispatch: AppDispatch;
        };

        export type RootState = ReturnType<typeof store.getState>
--------------------------------------------------------------------------------
export default store;
-Trong file store cua redux, doi voi ts, phai cau hinh them AppDispatch de khong bi bao loi:

    type Config = {
    dispatch: AppDispatch;
    };

- Cau Hinh them RootState de co them cham goi cac reduer mot cach de dang:
    export type RootState = ReturnType<typeof store.getState>
